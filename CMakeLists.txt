########################################################################
# Project setup
########################################################################
cmake_minimum_required(VERSION 2.6)
project(tsbe CXX)
enable_testing()

#select the release build type by default to get optimization flags
if(NOT CMAKE_BUILD_TYPE)
   set(CMAKE_BUILD_TYPE "Release")
   message(STATUS "Build type not specified: defaulting to release.")
endif(NOT CMAKE_BUILD_TYPE)
set(CMAKE_BUILD_TYPE ${CMAKE_BUILD_TYPE} CACHE STRING "")

if(CMAKE_COMPILER_IS_GNUCXX AND "${CMAKE_SYSTEM_PROCESSOR}" STREQUAL "x86_64")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -fPIC")
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_DEBUG} -fPIC")
endif()

########################################################################
# Setup Theron
########################################################################
add_subdirectory(Theron)
include_directories(${CMAKE_SOURCE_DIR}/Theron/Include)
if (NOT WIN32) #not on windows? tell theron headers to use boost
    add_definitions(-DTHERON_USE_BOOST_THREADS)
endif()

########################################################################
# Setup NumaNuma
########################################################################
list(APPEND CMAKE_MODULE_PATH ${CMAKE_SOURCE_DIR}/numanuma/cmake)
include_directories(${CMAKE_SOURCE_DIR}/numanuma/include)
find_package(NumaNuma)
message(STATUS "NUMANUMA_LIBRARIES: ${NUMANUMA_LIBRARIES}")

########################################################################
# Setup Boost
########################################################################
if(UNIX AND EXISTS "/usr/lib64")
    list(APPEND BOOST_LIBRARYDIR "/usr/lib64") #fedora 64-bit fix
endif(UNIX AND EXISTS "/usr/lib64")

set(BOOST_REQUIRED_COMPONENTS thread unit_test_framework)
if(MSVC)
    unset(BOOST_REQUIRED_COMPONENTS) #empty components list for static link
endif(MSVC)

SET(Boost_ADDITIONAL_VERSIONS
    "1.35.0" "1.35" "1.36.0" "1.36" "1.37.0" "1.37" "1.38.0" "1.38" "1.39.0" "1.39"
    "1.40.0" "1.40" "1.41.0" "1.41" "1.42.0" "1.42" "1.43.0" "1.43" "1.44.0" "1.44"
    "1.45.0" "1.45" "1.46.0" "1.46" "1.47.0" "1.47" "1.48.0" "1.48" "1.49.0" "1.49"
    "1.50.0" "1.50" "1.51.0" "1.51" "1.52.0" "1.52" "1.53.0" "1.53" "1.54.0" "1.54"
    "1.55.0" "1.55" "1.56.0" "1.56" "1.57.0" "1.57" "1.58.0" "1.58" "1.59.0" "1.59"
    "1.60.0" "1.60" "1.61.0" "1.61" "1.62.0" "1.62" "1.63.0" "1.63" "1.64.0" "1.64"
    "1.65.0" "1.65" "1.66.0" "1.66" "1.67.0" "1.67" "1.68.0" "1.68" "1.69.0" "1.69"

find_package(Boost COMPONENTS ${BOOST_REQUIRED_COMPONENTS})

########################################################################
# Local Include Dir
########################################################################
include_directories(${CMAKE_SOURCE_DIR}/include)

########################################################################
# Do subdirectories
########################################################################
add_subdirectory(include/tsbe)
add_subdirectory(lib)
add_subdirectory(tests)
